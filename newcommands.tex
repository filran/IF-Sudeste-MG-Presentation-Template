% Text emphasis
\newcommand{\emphasis}[1]{
    \centering
    \Large
    \textbf{#1}
}

% \pause + \item
\newcommand{\pitem}{\pause\item}

% Table of content highlight before a section
\newcommand{\outline}{
    \begin{frame}
    \frametitle{Outline}
    \tableofcontents[currentsection]
    \end{frame}
}

% Subsection overwrite
% oldsubsection
\let\oldsubsection\subsection
% overwriting \subsection
\renewcommand{\subsection}[1]{
    \oldsubsection{#1} % Original \subsection
    \begin{frame}
        \emphasis{#1}
    \end{frame}
}

% Subsubsection overwrite
% oldsubsubsection
\let\oldsubsubsection\subsubsection
% overwriting \subsubsection
\renewcommand{\subsubsection}[1]{
    \oldsubsubsection{#1} % Original \subsubsection
    \begin{frame}
        \emphasis{#1}
    \end{frame}
}
